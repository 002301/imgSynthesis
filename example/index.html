img-synthesis

图片合成应用

创建纯色图片

import IS from '../src/imgSynthesis.min.js';
let is = new IS(950,300,"#0381ff");
let img = new Image();
img.src = is.getImg();
document.querySelector(".drow").appendChild(img);

添加文字

import IS from '../src/imgSynthesis.min.js';
	let is2 = new IS(950,300, "#0381ff");
	is2.addTxt("有人吗？",100, 100,{
		w:256, h:0,//宽高
		size:'44px',//文字大小
		color:'#ffffff',//文字颜色
		writingMode:'td',//竖排显示
		align:'center' //对齐方式
	});
	//多文本添加
	is2.addTxts([['千山鸟飞绝', 100, 50],['万径人踪灭', 200, 150]]).then(e=>{
	let img2 = new Image();
	img2.src = is2.getImg();
	document.querySelector(".drow2").appendChild(img2);
	})
